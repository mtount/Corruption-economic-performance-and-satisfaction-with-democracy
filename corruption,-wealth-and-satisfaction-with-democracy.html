Load the data set

# Load the dataset
data <- read_dta("Release_v1_20230417_W5_merge_15.dta")

Filter the data set for the country code 4 (= South Korea)

data1 <- data %>%
  filter(country == 4)

Select variables for the analysis

# Now select the variables for the analysis 
data2 <- data1 %>% 
  dplyr::select(q99, q125, q1, 
                SE2, SE3_1, SE12, SE5A, q46, w)

Recoding of variables and specification of NAs (missing values)

# Now consider q125: perceived level of national corruption, and specify the NAs
data2$q125[data2$q125 == 7] <- NA
data2$q125[data2$q125 == 8] <- NA
data2$q125[data2$q125 == 9] <- NA
data2$q125[data2$q125 == -1] <- NA

# Now consider q99: satisfaction with democracy, specify the NA's and make it binary so that fairly and very satisfied = 1
data2$q99[data2$q99 == 7] <- NA
data2$q99[data2$q99 == 8] <- NA
data2$q99[data2$q99 == 9] <- NA

data2$q99[data2$q99 == 1] <- 1
data2$q99[data2$q99 == 2] <- 1
data2$q99[data2$q99 == 3] <- 0
data2$q99[data2$q99 == 4] <- 0

# Now consider SE2: gender, make it binary (There are no missing values) 
data2$SE2_dummy <- ifelse(data2$SE2 == 1, 0, 1) # where 1=female, 0=men

# Now consider SE12: subjective social status, and specify the NAs
data2$SE12[data2$SE12 == 98] <- NA
data2$SE12[data2$SE12 == 99] <- NA

# Now consider q46: interested in politics and specify the NAs
data2$q46[data2$q46 == 9] <- NA
data2$q46[data2$q46 == 8] <- NA
data2$q46[data2$q46 == 7] <- NA
data2$q46[data2$q46 == -1] <- NA

# Now consider q1: economic performance, specify the NAs
data2$q1[data2$q1 == 7] <- NA
data2$q1[data2$q1 == 8] <- NA
data2$q1[data2$q1 == 9] <- NA
data2$q1[data2$q1 == -1] <- NA

# Now consider SE5A: How many years of formal education and specify the NAs

Now remove all NA’s from the control, independent and dependent variables

In this way, we can compare the informationcriteria BIC / AIC for Model 1 (without control variables) and for Model 2 (with control variables)

data3 <- data2%>%filter(!is.na(q99 + q125 +q1 + q46 + SE12 +SE5A))

Now reverse the scale for two variables (perceived national economic performance + Political interest)

data4 <- data3 %>% mutate(q1 = case_when( # perceived national economic performance
  q1 == 1 ~ 5, # very good 
  q1 == 2 ~ 4, # Good
  q1 == 3 ~ 3, # So so
  q1 == 4 ~ 2, # Bad
  q1 == 5 ~ 1, #very bad 
  )) %>% 
  mutate(q46 = case_when( # Political Interest
    q46 == 1 ~ 4, # Very interested 
    q46 == 2 ~ 3, # somewhat interestes
    q46 == 3 ~ 2, # not very interested 
    q46 == 4 ~ 1)) # not at all interested

Regression analysis

# Without control variables 
model <- glm(q99 ~ q125 + q1, weights = w,
             data = data4 ,family = binomial)

texreg(model, custom.coef.names = c("(Intercept)", 
                                    "Perceived level of national corruption",
                                    "Perceived national economic performance"), 
       custom.model.names = ("Dependent variable: Satisfaction with democracy"), 
       caption = "Binary Logistic Regression Result without control variables")
# With control variables 
model1 <- glm(q99 ~ q125 +q1 + SE2_dummy + SE3_1 + SE5A + SE12 + q46, weights = w,
              data = data4 ,family = binomial)
texreg(model1, custom.coef.names = c("(Intercept)", 
                                    "Perceived level of national corruption",
                                    "Perceived national economic performance", 
                                    "Gender: Female", "Age", "Education", 
                                    "Subjective socioeconomic status", "Political interest"), 
       custom.model.names = ("Dependent variable: Satisfaction with democracy"), 
       caption = "Binary Logistic Regression Result")
texreg(list(model, model1), custom.coef.names = c("(Intercept)", 
                                    "Perceived level of national corruption",
                                    "Perceived national economic performance", 
                                    "Gender: Female", "Age", "Education", 
                                    "Subjective socioeconomic status", "Political interest"), 
            custom.model.names = c("Model 1: Satisfaction w. Democracy", 
                                   "Model 2: Satisfaction w. Democracy"), 
       caption = "Binary Logistic Regression Result")

Predicted probabilities

# Simulated predicted probabilites for corruption
model2 <- glm(q99 ~ q125 + SE2_dummy + SE3_1 + SE5A + SE12 + q46, weights = w,
              data = data4 ,family = binomial)

df_corruption <- predicts(model2, "1-4; mode; mean; median; median; median", sim.count = 1000, set.seed = 12139, type = "simulation")

# Plot 
t <- ggplot(df_corruption,
            aes(x = q125,
                y = mean, ymin = lower, ymax = upper)) +
  geom_pointrange() + theme_bw() +
  scale_y_continuous(labels = scales::percent) + ylab("Predicted Probability to be satisfied with democracy") + xlab ("Perceived national corruption") +
  theme(panel.grid.minor.x = element_blank()) +
  theme(axis.title.x = element_text(vjust = -1)) +
  theme(axis.title.y = element_text(vjust = +1)) + ggtitle("Predicted Probabilities: Perceived national corruption & Satisfaction with Democracy") + scale_x_discrete(limits = c("Hardly anyone is involved", "Not a lot of officials are corrupt", "Most officials are corrupt", "Almost everyone is corrupt"))
